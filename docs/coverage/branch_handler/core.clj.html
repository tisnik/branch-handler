106  (defn read-branch-list-from-repo
107    "Read list of all branches from selected repository."
108    [workdir group reponame]
109    (let [repodir (repo-workdir workdir group reponame)]
110      (log/info "Reading branches from repository" repodir)
111      (->> (shell/sh "git"
112                     "--git-dir"
113                     (repo-workdir workdir group reponame)
114                     "branch")
115           :out
116           string/split-lines
117           (map #(subs % 2)))))

119  (defn clone-or-fetch-mirror-repo
120    "Clone the repository or fetch data if directory with clone already exist."
121    [action]
122    (log/info "Cloning or fetching repository")
123    (let [reponame       (:name action)
124          group          (:group action)
125          repository-url (:url action)]
126      (log/info (repo-workdir workdir group reponame))
127      (if (repodir-exists? workdir group reponame)
128        (fetch-mirror-repo workdir group reponame repository-url)
129        (clone-mirror-repo workdir group reponame repository-url))))

150  (defn parse-job-name
151    "Parse the job name, try to get group, repository, and branch from it."
152    [branch-job-prefix job-name]
153    (let [parsed (clojure.string/split job-name #"\+")]
154      (if (= (count parsed) 3)
155        {:name       job-name,
156         :group      (subs (clojure.string/trim (first parsed))
157                           (count branch-job-prefix)),
158         :repository (clojure.string/trim (second parsed)),
159         :branch     (clojure.string/trim (nth parsed 2))})))
160  
161  (defn parse-job-names
162    "Parse all job names, try to get group, repository, and branch from them."
163    [branch-job-prefix job-names]
164    (for [job-name job-names]
165         (parse-job-name branch-job-prefix job-name)))

168    "Create or delete Jenkins job according to selected action."
169    [jenkins-url job-list-command branch-job-prefix action]
170    (log/info "Create-or-delete-jenkins-job")
171    (let [reponame       (:name action)
172          group          (:group action)
173          jobs-for-branches (read-jobs-for-branches jenkins-url
174                                                    job-list-command
175                                                    branch-job-prefix)
176          jobs-info      (parse-job-names branch-job-prefix jobs-for-branches)
177          branch-list-in-repo (read-branch-list-from-repo workdir group reponame)]
178      jobs-info))

185  (defn action-consumer
186    "Consumer that process messages dequed from action queue."
187    [actions-queue]
188    (while true
189           (when-let [action (.take actions-queue)] ; blocking operation
190             (log/info "Started action for repository" action)
191             (clone-or-fetch-mirror-repo action)
192             (create-or-delete-jenkins-job jenkins-url
193                                           job-list-command
194                                           branch-job-prefix
195                                           action)
196             (start-jenkins-jobs jenkins-url action)
197             (log/info "Finished action for repository" action))))

211  (def empty-SHA
212    "Default SHA used to detect operation that needs to be performed."
213    "0000000000000000000000000000000000000000")
214  
215  (defn get-operation
216    "Get the next operation that needs to be planned."
217    [before after]
218    (if (= before empty-SHA)
219      (if (= after empty-SHA)
220          :unknown
221          :branch-created)
222      (if (= after empty-SHA)
223          :branch-deleted
224          :branch-updated)))

226  (defn get-branch
227    "Retrieve branch name for given Git reference."
228    [git-ref]
229    (try (subs git-ref (inc (.lastIndexOf git-ref "/")))
230         (catch Exception e
231                nil)))

237  (defn api-call-handler
238    "Handler for all API calls."
239    [request]
240    (let [gitlab-info      (parse-gitlab-info request)
241          repository-url   (-> gitlab-info :repository :url)
242          repository-name  (-> gitlab-info :repository :name)
243          repository-group (-> gitlab-info :project :namespace)
244          branch           (get-branch (:ref gitlab-info))
245          before           (:before gitlab-info)
246          after            (:after gitlab-info)
247          operation        (get-operation before after)]
248      (if verbose
249        (clojure.pprint/pprint gitlab-info))
250      (log/info "repository" repository-name)
251      (log/info "URL"        repository-url)
252      (log/info "group"      repository-group)
253      (log/info "branch"     branch)
254      (log/info "before"     before)
255      (log/info "after"      after)
256      (log/info "operation"  operation)
257      (if (contains? #{:branch-created :branch-deleted :branch-updated} operation)
258        (let [action {:action :clone-or-fetch-repository,
259                      :name    repository-name,
260                      :group   repository-group,
261                      :url     repository-url,
262                      :original-data gitlab-info}]
263          (new-action actions-queue action))) ; add new action into the queue that
264                                              ; will serialize it
265    ))

268    "Handler that is called by Ring for all requests received from user(s)."
269    [request]

271    (if (= [:post "/"] [(:request-method request) (:uri request)])
272      (api-call-handler request))
273    (send-response ""))
